\chapter{Time Complexity}
\section{$\mathbf{P}$}
\begin{definition}
  We define
  \begin{equation*}
    \mathbf{P} = \bigcup_{k \in \mathbb{N}} \mathbf{TIME}(n^k).
  \end{equation*}
\end{definition}

\section{$\mathbf{NP}$}
\begin{definition}
  We define
  \begin{equation*}
    \mathbf{NP} = \bigcup_{k \in \mathbb{N}} \mathbf{NTIME}(n^k).
  \end{equation*}
\end{definition}
